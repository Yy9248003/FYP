## 🎨 设计系统使用

### 引入设计系统

设计系统已自动引入，无需额外配置。

### 使用CSS变量

```vue
<style scoped>
.my-component {
  /* 使用主色调 */
  background: var(--primary-gradient);
  color: var(--text-primary);
  
  /* 使用间距 */
  padding: var(--spacing-lg);
  margin: var(--spacing-md);
  
  /* 使用圆角 */
  border-radius: var(--border-radius-lg);
  
  /* 使用阴影 */
  box-shadow: var(--shadow-md);
  
  /* 使用过渡 */
  transition: all var(--transition-base);
}
</style>
```

### 使用预定义类

```vue
<template>
```

## 🎯 交互组件使用

### 1. 增强加载组件

```vue
<template>
  <EnhancedLoading
    :visible="loading"
    :text="loadingText"
    :show-progress="showProgress"
    :progress="progress"
    :closable="false"
    @close="handleClose" />
</template>

<script>
import EnhancedLoading from '@/components/EnhancedLoading.vue'

export default {
  components: {
    EnhancedLoading
  },
  data() {
    return {
      loading: false,
      loadingText: '加载中...',
      showProgress: false,
      progress: 0
    }
  },
  methods: {
    async loadData() {
      this.loading = true
      this.showProgress = true
      
      try {
        // 模拟进度
        for (let i = 0; i <= 100; i += 10) {
          this.progress = i
          await new Promise(resolve => setTimeout(resolve, 200))
        }
        
        // 加载数据
        // ...
      } finally {
        this.loading = false
        this.showProgress = false
      }
    },
    handleClose() {
      this.loading = false
    }
  }
}
</script>
```

### 2. 增强提示组件

```vue
<template>
  <EnhancedToast
    :visible="showToast"
    :type="toastType"
    :title="toastTitle"
    :message="toastMessage"
    :duration="3000"
    :closable="true"
    @close="showToast = false" />
</template>

<script>
import EnhancedToast from '@/components/EnhancedToast.vue'

export default {
  components: {
    EnhancedToast
  },
  data() {
    return {
      showToast: false,
      toastType: 'success',
      toastTitle: '成功',
      toastMessage: '操作成功'
    }
  },
  methods: {
    showSuccessToast(message, title = '成功') {
      this.toastType = 'success'
      this.toastTitle = title
      this.toastMessage = message
      this.showToast = true
    },
    showErrorToast(message, title = '错误') {
      this.toastType = 'error'
      this.toastTitle = title
      this.toastMessage = message
      this.showToast = true
    }
  }
}
</script>
```

## 🛠️ UI辅助函数使用

### 引入辅助函数

```javascript
import { 
  showSuccess, 
  showError, 
  formatTime, 
  copyToClipboard 
} from '@/utils/ui-helpers.js'
```

### 提示函数

```javascript
// 成功提示
showSuccess('操作成功', '成功', 3000)

// 错误提示
showError('操作失败', '错误', 3000)

// 警告提示
showWarning('请注意', '警告', 3000)

// 信息提示
showInfo('这是一条信息', '提示', 3000)
```

### 确认对话框

```javascript
import { showConfirm } from '@/utils/ui-helpers.js'

showConfirm(
  '确认删除',
  '确定要删除这条记录吗？删除后无法恢复。',
  () => {
    // 确认操作
    console.log('确认删除')
  },
  () => {
    // 取消操作
    console.log('取消删除')
  }
)
```

### 时间格式化

```javascript
import { formatTime, formatRelativeTime } from '@/utils/ui-helpers.js'

// 格式化时间
const formatted = formatTime('2024-01-01 12:00:00', 'YYYY-MM-DD')
// 结果: "2024-01-01"

// 相对时间
const relative = formatRelativeTime('2024-01-01 12:00:00')
// 结果: "3天前" 或 "2小时前" 等
```

### 复制到剪贴板

```javascript
import { copyToClipboard } from '@/utils/ui-helpers.js'

// 复制文本
copyToClipboard('要复制的文本')
  .then(success => {
    if (success) {
      console.log('复制成功')
    }
  })
```

### 防抖节流

```javascript
import { debounceDecorator, throttleDecorator } from '@/utils/ui-helpers.js'

// 防抖
const debouncedSearch = debounceDecorator((keyword) => {
  console.log('搜索:', keyword)
}, 300)

// 节流
const throttledScroll = throttleDecorator(() => {
  console.log('滚动事件')
}, 100)
```
