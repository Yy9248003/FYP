### 1. 密码加密存储（最紧急！）

**问题**: 当前密码明文存储，严重安全漏洞

**位置**: `source/server/app/views.py:98`

**当前代码**:
```python
if user.passWord == passWord:  # 明文比较
```

**优化方案**:
```python
from django.contrib.auth.hashers import make_password, check_password

# 登录验证
def login(request):
    userName = request.POST.get('userName')
    passWord = request.POST.get('passWord')
    
    user = models.Users.objects.filter(userName=userName).first()
    if user:
        # 检查密码是否为旧明文格式，如果是则迁移
        if len(user.passWord) < 50:  # 旧明文密码
            if user.passWord == passWord:
                # 迁移为加密密码
                user.passWord = make_password(passWord)
                user.save()
                # 继续登录流程
            else:
                return BaseView.warn('用户密码输入错误')
        else:
            # 使用加密验证
            if check_password(passWord, user.passWord):
                # 登录成功
            else:
                return BaseView.warn('用户密码输入错误')
```

**迁移脚本**:
```python
# source/server/migrate_passwords.py
from django.contrib.auth.hashers import make_password
from app import models

def migrate_passwords():
    """迁移所有明文密码为加密密码"""
    users = models.Users.objects.all()
    for user in users:
        # 检查是否为明文（长度小于50）
        if len(user.passWord) < 50:
            # 保存明文密码
            plain_password = user.passWord
            # 加密并更新
            user.passWord = make_password(plain_password)
            user.save()
            print(f"已迁移用户: {user.userName}")
```

**影响范围**:
- `views.py` - login方法
- `views.py` - updUserPwd方法
- `views.py` - addInfo方法（创建用户时）
- 需要数据迁移脚本

### 3. API限流保护

**问题**: 缺少API限流，可能被恶意请求攻击

**优化方案**:
```python
# source/server/comm/rate_limit.py
from django.core.cache import cache
from django.http import JsonResponse
import time

def rate_limit(max_requests=100, period=60, key_prefix='api'):
    """
    API限流装饰器
    max_requests: 最大请求数
    period: 时间窗口（秒）
    """
    def decorator(func):
        def wrapper(request, *args, **kwargs):
            # 获取客户端IP
            client_ip = request.META.get('REMOTE_ADDR', 'unknown')
            # 获取用户ID（如果已登录）
            user_id = cache.get(request.POST.get('token') or request.GET.get('token'))
            # 构建限流key
            if user_id:
                key = f'{key_prefix}:user:{user_id}'
            else:
                key = f'{key_prefix}:ip:{client_ip}'
            
            # 获取当前请求数
            current = cache.get(key, 0)
            
            if current >= max_requests:
                return JsonResponse({
                    'code': 429,
                    'msg': f'请求过于频繁，请{period}秒后再试'
                }, status=429)
            
            # 增加计数
            cache.set(key, current + 1, period)
            
            return func(request, *args, **kwargs)
        return wrapper
    return decorator
```

**使用示例**:
```python
from comm.rate_limit import rate_limit

@rate_limit(max_requests=60, period=60)
def login(request):
    # 登录逻辑
    pass
```

```python
# source/server/app/views.py

# 在Messages模型中添加附件字段（需要迁移）
# attachment = models.FileField(upload_to='messages/attachments/', null=True, blank=True)

@staticmethod
def uploadAttachment(request):
    """上传附件"""
    try:
        if 'file' not in request.FILES:
            return BaseView.error('未选择文件')
        
        file = request.FILES['file']
        # 文件大小限制（10MB）
        if file.size > 10 * 1024 * 1024:
            return BaseView.error('文件大小不能超过10MB')
        
        # 保存文件
        from django.core.files.storage import default_storage
        file_path = default_storage.save(
            f'messages/attachments/{file.name}',
            file
        )
        
        return BaseView.successData({
            'file_path': file_path,
            'file_name': file.name,
            'file_size': file.size
        })
    except Exception as e:
        return BaseView.error(f'上传失败: {str(e)}')

@staticmethod
def downloadAttachment(request):
    """下载附件"""
    try:
        file_path = request.GET.get('file_path')
        if not file_path:
            return BaseView.error('文件路径不能为空')
        
        from django.http import FileResponse
        from django.core.files.storage import default_storage
        
        if default_storage.exists(file_path):
            file = default_storage.open(file_path)
            response = FileResponse(file)
            response['Content-Disposition'] = f'attachment; filename="{file_path.split("/")[-1]}"'
            return response
        else:
            return BaseView.error('文件不存在')
    except Exception as e:
        return BaseView.error(f'下载失败: {str(e)}')
```

### 5. 统一错误处理机制

**问题**: 错误处理分散，不够统一

**优化方案**:
```python
# source/server/comm/error_handler.py
import logging
from django.http import JsonResponse
from django.core.exceptions import ValidationError, PermissionDenied

logger = logging.getLogger(__name__)

class ErrorHandler:
    """统一错误处理"""
    
    @staticmethod
    def handle_exception(func):
        """异常处理装饰器"""
        def wrapper(request, *args, **kwargs):
            try:
                return func(request, *args, **kwargs)
            except ValidationError as e:
                logger.warning(f"验证错误: {str(e)}")
                return JsonResponse({
                    'code': 400,
                    'msg': f'参数验证失败: {str(e)}'
                }, status=400)
            except PermissionDenied as e:
                logger.warning(f"权限错误: {str(e)}")
                return JsonResponse({
                    'code': 403,
                    'msg': '权限不足'
                }, status=403)
            except Exception as e:
                logger.error(f"系统错误: {str(e)}", exc_info=True)
                return JsonResponse({
                    'code': 500,
                    'msg': '系统错误，请稍后重试'
                }, status=500)
        return wrapper
```

**使用示例**:
```python
from comm.error_handler import ErrorHandler

class SysView(BaseView):
    @ErrorHandler.handle_exception
    def login(request):
        # 登录逻辑
        pass
```
